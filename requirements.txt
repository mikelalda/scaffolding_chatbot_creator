import tkinter as tk
from tkinter import messagebox

class ChatbotScaffoldingApp:
    def __init__(self, master):
        self.master = master
        master.title("Chatbot Scaffolding Guide")

        self.steps = [
            "Step 1: Install Python and necessary libraries.",
            "Step 2: Create a new Python file for your chatbot.",
            "Step 3: Define your chatbot's intents.",
            "Step 4: Create a function to handle user input.",
            "Step 5: Implement a simple response mechanism.",
            "Step 6: Test your chatbot."
        ]

        self.current_step = 0

        self.label = tk.Label(master, text=self.steps[self.current_step], wraplength=400)
        self.label.pack(pady=20)

        self.next_button = tk.Button(master, text="Next Step", command=self.next_step)
        self.next_button.pack(pady=10)

        self.prev_button = tk.Button(master, text="Previous Step", command=self.prev_step)
        self.prev_button.pack(pady=10)

        self.finish_button = tk.Button(master, text="Finish", command=self.finish)
        self.finish_button.pack(pady=10)

        self.prev_button.config(state=tk.DISABLED)  # Disable previous button on the first step

    def next_step(self):
        if self.current_step < len(self.steps) - 1:
            self.current_step += 1
            self.label.config(text=self.steps[self.current_step])
            self.prev_button.config(state=tk.NORMAL)  # Enable previous button
        if self.current_step == len(self.steps) - 1:
            self.next_button.config(state=tk.DISABLED)  # Disable next button on the last step

    def prev_step(self):
        if self.current_step > 0:
            self.current_step -= 1
            self.label.config(text=self.steps[self.current_step])
            self.next_button.config(state=tk.NORMAL)  # Enable next button
        if self.current_step == 0:
            self.prev_button.config(state=tk.DISABLED)  # Disable previous button on the first step

    def finish(self):
        messagebox.showinfo("Finished", "You have completed the chatbot scaffolding guide!")
        self.master.quit()

if __name__ == "__main__":
    root = tk.Tk()
    app = ChatbotScaffoldingApp(root)
    root.mainloop()
